21uint64_t cwist_get_available_ram(void) {
22 // Attempt 1: sysinfo() (Fastest, standard libc)
23 struct sysinfo si;
24 if (sysinfo(&si) == 0) {
25 // freeram is explicitly free.
26 // bufferram and sharedram might be reclaimable but let's be conservative.
27 // On modern Linux kernels, "MemAvailable" in /proc/meminfo is better,
28 // but sysinfo is a good rough estimate multiplied by mem_unit.
29 return (uint64_t)si.freeram * (uint64_t)si.mem_unit;
30 }
31
32 // Attempt 2: /proc/meminfo (More accurate "Available")
33 FILE *fp = fopen("/proc/meminfo", "r");
34 if (fp) {
35 char line[256];
36 uint64_t mem_avail_kb = 0;
37 while (fgets(line, sizeof(line), fp)) {
38 if (sscanf(line, "MemAvailable: %lu kB", &mem_avail_kb) == 1) {
39 fclose(fp);
40 return mem_avail_kb * 1024;
41 }
42 }
43 fclose(fp);
44 }
45
46 return 0; // Unknown
47}
