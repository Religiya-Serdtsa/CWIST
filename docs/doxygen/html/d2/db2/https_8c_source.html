50cwist_error_t cwist_https_init_context(cwist_https_context **ctx, const char *cert_path, const char *key_path) {
51 cwist_error_t err = make_error(CWIST_ERR_INT16);
52
53 if (!ctx || !cert_path || !key_path) {
54 err.error.err_i16 = -1;
55 return err;
56 }
57
58 // Initialize OpenSSL
59 SSL_load_error_strings();
60 OpenSSL_add_ssl_algorithms();
61
62 const SSL_METHOD *method = TLS_server_method();
63 SSL_CTX *ssl_ctx = SSL_CTX_new(method);
64 if (!ssl_ctx) {
65 return make_ssl_error("Unable to create SSL context");
66 }
67
68 if (SSL_CTX_set_min_proto_version(ssl_ctx, TLS1_2_VERSION) != 1) {
69 SSL_CTX_free(ssl_ctx);
70 return make_ssl_error("Unable to enforce TLS minimum version");
71 }
72 SSL_CTX_set_options(ssl_ctx, SSL_OP_NO_COMPRESSION);
73
74 // Load Cert and Key
75 if (SSL_CTX_use_certificate_file(ssl_ctx, cert_path, SSL_FILETYPE_PEM) <= 0) {
76 SSL_CTX_free(ssl_ctx);
77 return make_ssl_error("Unable to load certificate");
78 }
79
80 if (SSL_CTX_use_PrivateKey_file(ssl_ctx, key_path, SSL_FILETYPE_PEM) <= 0) {
81 SSL_CTX_free(ssl_ctx);
82 return make_ssl_error("Unable to load private key");
83 }
84
85 // Verify key matches cert
86 if (!SSL_CTX_check_private_key(ssl_ctx)) {
87 SSL_CTX_free(ssl_ctx);
88 return make_ssl_error("Private key does not match certificate");
89 }
90
91 *ctx = (cwist_https_context*)cwist_alloc(sizeof(cwist_https_context));
92 if (!*ctx) {
93 SSL_CTX_free(ssl_ctx);
94 err.error.err_i16 = -1;
95 return err;
96 }
97 (*ctx)->ctx = ssl_ctx;
98
99 err.error.err_i16 = 0; // Success
100 return err;

125 cwist_error_t err = make_error(CWIST_ERR_INT16);

157 if (!(*conn)->read_buf) {

161 err.error.err_i16 = -1;
162 return err;
163 }
164 (*conn)->buf_len = 0;
165 (*conn)->read_buf[0] = '\0';

230 size_t body_received = total_received - header_len;

232 if (req->content_length > 0) {

234 cwist_http_request_destroy(req);
235 return NULL;

241 return NULL;
242 }

255 }

266 }

268 }

271 cwist_free(body);

279 }

283 conn->buf_len = body_received;

285 conn->buf_len = 0;

287 }

299cwist_error_t cwist_https_send_response(cwist_https_connection *conn, cwist_http_response *res) {
300 cwist_error_t err = make_error(CWIST_ERR_INT16);
301
302 if (!conn || !conn->ssl || !res) {
303 err.error.err_i16 = -1;
304 return err;

306
307 // 1. Serialize using existing HTTP logic
308 cwist_sstring *response_str = cwist_http_stringify_response(res);
309 if (!response_str) {

311 return err;
312 }
313

316 int left = (int)response_str->size;
317 int total_sent = 0;
318
319 err.error.err_i16 = 0; // Assume success initially
320
321 while (left > 0) {
322 int sent = SSL_write(conn->ssl, p, left);
323 if (sent <= 0) {
324 int ssl_err = SSL_get_error(conn->ssl, sent);
325 if (ssl_err == SSL_ERROR_WANT_WRITE || ssl_err == SSL_ERROR_WANT_READ) {
326 continue; // Retry

329 break;
330 }
331 p += sent;
332 left -= sent;
333 total_sent += sent;
334 }
335
336 cwist_sstring_destroy(response_str);
337 return err;
