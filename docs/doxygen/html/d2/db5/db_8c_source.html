30static uint64_t cwist_db_integer_root(uint64_t value) {
31 if (value == 0) return 0;
32 uint64_t x = value;
33 uint64_t y = (x + 1) >> 1;
34 while (y < x) {
35 x = y;
36 y = (x + value / x) >> 1;
37 }
38 return x;
39}

57static bool cwist_db_match_keyword(const char *sql, size_t pos, const char *kw) {
58 size_t i = 0;
59 while (kw[i]) {
60 char current = sql[pos + i];
61 if (!current) return false;
62 if ((char)toupper((unsigned char)current) != kw[i]) {
63 return false;
64 }
65 ++i;
66 }
67 if (pos > 0) {
68 char prev = sql[pos - 1];
69 if (cwist_db_is_alpha(prev)) return false;
70 }
71 char next = sql[pos + i];
72 if (cwist_db_is_alpha(next)) return false;
73 return true;
74}

81static cwist_db_plan_hint cwist_db_analyze_sql(const char *sql) {
82 cwist_db_plan_hint hint = {0, 0, 0};
83 if (!sql) return hint;
84 size_t len = strlen(sql);
85 uint64_t accum = 0xcbf29ce484222325ULL;
86 uint32_t depth = 0;
87 for (size_t i = 0; i < len; ++i) {
88 char upper = (char)toupper((unsigned char)sql[i]);
89 accum ^= (uint64_t)upper;
90 accum *= 0x100000001b3ULL;
91 if (upper == '(') depth++;
92 else if (upper == ')' && depth > 0) depth--;
93
94 if (upper == 'J' && cwist_db_match_keyword(sql, i, "JOIN")) {
95 hint.join_count++;
96 } else if (upper == 'W' && cwist_db_match_keyword(sql, i, "WHERE")) {
97 hint.predicate_count++;
98 } else if (upper == 'A' && cwist_db_match_keyword(sql, i, "AND")) {
99 hint.predicate_count++;
100 } else if (upper == 'O' && cwist_db_match_keyword(sql, i, "OR")) {
101 hint.predicate_count++;
102 }
103 }
104 hint.symbolic_weight = accum ^ ((uint64_t)depth << 48);
105 return hint;
106}

113static void cwist_db_apply_hint(sqlite3 *conn, const cwist_db_plan_hint *hint) {
114 if (!conn || !hint) return;
115 uint64_t complexity = hint->symbolic_weight +
116 ((uint64_t)hint->join_count * 131ULL) +
117 ((uint64_t)hint->predicate_count * 53ULL) + 1ULL;
118 uint64_t root = cwist_db_integer_root(complexity);
119 int busy_ms = 50 + (int)(root % 350);
120 sqlite3_busy_timeout(conn, busy_ms);
121 if (hint->join_count >= 2) {
122 sqlite3_exec(conn, "PRAGMA temp_store = MEMORY;", NULL, NULL, NULL);
123 }
124 if (hint->predicate_count >= 3) {
125 sqlite3_exec(conn, "PRAGMA cache_spill = 0;", NULL, NULL, NULL);
126 }
127}

237static int query_callback(void *data, int argc, char **argv, char **azColName) {

303/*
304 * Build an INSERT statement from a cJSON object.
305 * Returns a heap-allocated (sqlite3_malloc) SQL string, or NULL on failure.
306 * The caller must free it with sqlite3_free().
307 *
308 * String values are escaped with %Q; numbers, booleans and nulls are
309 * rendered directly. Nested objects/arrays are serialised to a JSON
310 * string so they can be stored as TEXT.
311 */
318static char *build_insert_sql(const char *table, const cJSON *obj) {
319 /* Collect column/value fragments. */
320 int count = cJSON_GetArraySize((cJSON *)obj);
321 if (count <= 0) return NULL;
322
323 /* Worst-case sizes – we'll sqlite3_mprintf each fragment individually. */
324 char *cols_buf = (char *)sqlite3_malloc(count * 64 + 16);
325 char *vals_buf = (char *)sqlite3_malloc(count * 256 + 16);
326 if (!cols_buf || !vals_buf) {
327 sqlite3_free(cols_buf);
328 sqlite3_free(vals_buf);
329 return NULL;
330 }
331 cols_buf[0] = '\0';
332 vals_buf[0] = '\0';
333
334 const cJSON *child = obj->child;
335 bool first = true;
336 while (child) {
337 /* Column name – double-quoted for safety */
338 char col_frag[128];
339 snprintf(col_frag, sizeof(col_frag),
340 first ? "\"%s\"" : ",\"%s\"", child->string);
341 size_t cols_cap = (size_t)(count * 64 + 16);
342 size_t cols_used = strlen(cols_buf);
343 if (cols_used < cols_cap - 1)
344 strncat(cols_buf, col_frag, cols_cap - 1 - cols_used);
345
346 /* Value */
347 char *val_frag = NULL;
348 if (cJSON_IsString(child)) {
349 val_frag = sqlite3_mprintf(first ? "%Q" : ",%Q",
350 child->valuestring);
351 } else if (cJSON_IsNumber(child)) {
352 val_frag = sqlite3_mprintf(first ? "%g" : ",%g",
353 child->valuedouble);
354 } else if (cJSON_IsTrue(child)) {
355 val_frag = sqlite3_mprintf(first ? "1" : ",1");
356 } else if (cJSON_IsFalse(child)) {
357 val_frag = sqlite3_mprintf(first ? "0" : ",0");
358 } else if (cJSON_IsNull(child)) {
359 val_frag = sqlite3_mprintf(first ? "NULL" : ",NULL");
360 } else {
361 /* Object / Array → store as JSON text */
362 char *serialised = cJSON_PrintUnformatted(child);
363 if (serialised) {
364 val_frag = sqlite3_mprintf(first ? "%Q" : ",%Q", serialised);
365 free(serialised);
366 }
367 }
368
369 if (!val_frag) {
370 sqlite3_free(cols_buf);
371 sqlite3_free(vals_buf);
372 return NULL;
373 }
374 size_t vals_cap = (size_t)(count * 256 + 16);
375 size_t vals_used = strlen(vals_buf);
376 if (vals_used < vals_cap - 1)
377 strncat(vals_buf, val_frag, vals_cap - 1 - vals_used);
378 sqlite3_free(val_frag);
379
380 first = false;
381 child = child->next;
382 }
383
384 char *sql = sqlite3_mprintf("INSERT INTO \"%s\" (%s) VALUES (%s);",
385 table, cols_buf, vals_buf);
386 sqlite3_free(cols_buf);
387 sqlite3_free(vals_buf);
388 return sql;
389}

400cwist_error_t cwist_db_insert_healed(cwist_db *db, const char *table,
401 const char *json_str,
402 const cwist_schema_t *schema,
403 const cwist_heal_config_t *heal_cfg) {
404 cwist_error_t err = make_error(CWIST_ERR_INT16);
405 err.error.err_i16 = -1;
406
407 if (!db || !table || !json_str) return err;
408
409 /* Build effective heal config (merge schema if not already set). */
410 cwist_heal_config_t effective_cfg;
411 if (heal_cfg) {
412 effective_cfg = *heal_cfg;
413 } else {
414 memset(&effective_cfg, 0, sizeof(effective_cfg));
415 effective_cfg.threshold = 0.8;
416 }
417 if (schema && !effective_cfg.schema) effective_cfg.schema = schema;
418
419 /* Step 1: heal the JSON. */
420 cwist_heal_result_t healed = cwist_json_heal(json_str, &effective_cfg);
421 if (!healed.json) {
422 fprintf(stderr, "[CWIST-DB] insert_healed: healing failed for table '%s'\n",
423 table);
424 cwist_heal_result_free(&healed);
425 return err;
426 }
427
428 /* Step 2: strict Zod validation if a schema is provided. */
429 if (schema) {
430 cJSON *validated = NULL;
431 cwist_zod_result_t zod = cwist_zod_parse(healed.json, schema, &validated);
432 if (!zod.valid) {
433 fprintf(stderr,
434 "[CWIST-DB] insert_healed: Zod validation failed for table '%s':\n",
435 table);
436 cwist_zod_print_errors(&zod);
437 cwist_heal_result_free(&healed);
438 return err;
439 }
440 if (validated) cJSON_Delete(validated);
441 }
442
443 /* Step 3: parse the healed JSON and build INSERT SQL. */
444 cJSON *obj = cJSON_Parse(healed.json);
445 cwist_heal_result_free(&healed);
446 if (!obj || !cJSON_IsObject(obj)) {
447 if (obj) cJSON_Delete(obj);
448 return err;
449 }
450
451 char *sql = build_insert_sql(table, obj);
452 cJSON_Delete(obj);
453 if (!sql) return err;
454
455 /* Step 4: execute. */
456 err = cwist_db_exec(db, sql);
457 sqlite3_free(sql);
458 return err;
459}

473cwist_error_t cwist_db_query_strict(cwist_db *db, const char *sql, cJSON **result,
474 const cwist_schema_t *schema) {
475 /* When no schema is given, behave identically to cwist_db_query(). */
476 if (!schema) return cwist_db_query(db, sql, result);
477
478 cJSON *all = NULL;
479 cwist_error_t err = cwist_db_query(db, sql, &all);
480 if (err.errtype == CWIST_ERR_JSON || !all) return err;
481
482 cJSON *conforming = cJSON_CreateArray();
483 if (!conforming) {
484 cJSON_Delete(all);
485 err.error.err_i16 = -1;
486 return err;
487 }
488
489 cJSON *row = NULL;
490 cJSON_ArrayForEach(row, all) {
491 cwist_zod_result_t zod = cwist_zod_validate(row, schema);
492 if (zod.valid) {
493 cJSON *clone = cJSON_Duplicate(row, 1);
494 if (clone) cJSON_AddItemToArray(conforming, clone);
495 }
496 }
497
498 cJSON_Delete(all);
499 *result = conforming;
500
501 err = make_error(CWIST_ERR_INT16);
502 err.error.err_i16 = 0;
503 return err;
504}
