22static int aes256_encrypt(const unsigned char *key,
23 const unsigned char *iv,
24 const unsigned char *in, size_t in_len,
25 unsigned char *out) {
26 EVP_CIPHER_CTX *ctx = EVP_CIPHER_CTX_new();
27 if (!ctx) return -1;
28
29 int ok = 1;
30 int len1 = 0, len2 = 0;
31
32 ok &= EVP_EncryptInit_ex(ctx, EVP_aes_256_cbc(), NULL, key, iv);
33 ok &= EVP_EncryptUpdate(ctx, out, &len1, in, (int)in_len);
34 ok &= EVP_EncryptFinal_ex(ctx, out + len1, &len2);
35
36 EVP_CIPHER_CTX_free(ctx);
37 if (!ok) return -1;
38 return len1 + len2;
39}

69unsigned char *cwist_db_crypt_seal(const cwist_db_crypt_ctx_t *ctx,
70 const unsigned char *sqlite_bytes,
71 size_t sqlite_len,
72 size_t *out_len) {
73 if (!ctx || !sqlite_bytes || !out_len) return NULL;
74
75 /* 1. Generate random DEK and IVs. */
76 unsigned char dek[CWIST_DB_CRYPT_KEY_LEN];
77 unsigned char kek_iv[CWIST_DB_CRYPT_IV_LEN];
78 unsigned char dek_iv[CWIST_DB_CRYPT_IV_LEN];
79
80 if (RAND_bytes(dek, sizeof(dek)) != 1 ||
81 RAND_bytes(kek_iv, sizeof(kek_iv)) != 1 ||
82 RAND_bytes(dek_iv, sizeof(dek_iv)) != 1) {
83 return NULL;
84 }
85
86 /* 2. Wrap the DEK: encrypt ( dek_iv || dek ) with KEK.
87 * Plaintext: 16 + 32 = 48 bytes → cipher: 48 bytes (no padding needed,
88 * but EVP adds a full block when input is block-aligned, so we use 48
89 * and disable padding via a temp context).
90 * To keep it simple we just encrypt 48 bytes of dek_iv||dek normally;
91 * EVP pads to 64. We'll store all 64 for safe round-trip. */
92 unsigned char dek_plain[CWIST_DB_CRYPT_IV_LEN + CWIST_DB_CRYPT_KEY_LEN];
93 memcpy(dek_plain, dek_iv, CWIST_DB_CRYPT_IV_LEN);
94 memcpy(dek_plain + CWIST_DB_CRYPT_IV_LEN, dek, CWIST_DB_CRYPT_KEY_LEN);
95
96 /* Encrypted DEK blob: 48 plaintext bytes → 64 ciphertext bytes. */
97 unsigned char enc_dek[CWIST_DB_CRYPT_IV_LEN + CWIST_DB_CRYPT_KEY_LEN + 16];
98 int enc_dek_len = aes256_encrypt(ctx->kek, kek_iv,
99 dek_plain, sizeof(dek_plain),
100 enc_dek);
101 if (enc_dek_len <= 0) return NULL;
102
103 /* 3. Encrypt SQLite bytes with DEK. */
104 size_t ct_max = sqlite_len + 16 + 1; /* extra block for padding */
105 unsigned char *ct = (unsigned char *)malloc(ct_max);
106 if (!ct) return NULL;
107
108 int ct_len = aes256_encrypt(dek, dek_iv, sqlite_bytes, sqlite_len, ct);
109 if (ct_len <= 0) {
110 free(ct);
111 return NULL;
112 }
113
114 /* 4. Assemble blob:
115 * magic(4) + ver(1) + kek_iv(16) + enc_dek(enc_dek_len)
116 * + dek_iv(16) + ptlen(8) + ct(ct_len)
117 */
118 size_t hdr_fixed = 4 + 1 + CWIST_DB_CRYPT_IV_LEN + (size_t)enc_dek_len
119 + CWIST_DB_CRYPT_IV_LEN + 8;
120 size_t total = hdr_fixed + (size_t)ct_len;
121
122 unsigned char *blob = (unsigned char *)malloc(total);
123 if (!blob) {
124 free(ct);
125 return NULL;
126 }
127
128 size_t off = 0;
129 memcpy(blob + off, MAGIC, 4); off += 4;
130 blob[off] = VERSION; off += 1;
131 memcpy(blob + off, kek_iv, CWIST_DB_CRYPT_IV_LEN); off += CWIST_DB_CRYPT_IV_LEN;
132 memcpy(blob + off, enc_dek, (size_t)enc_dek_len); off += (size_t)enc_dek_len;
133 memcpy(blob + off, dek_iv, CWIST_DB_CRYPT_IV_LEN); off += CWIST_DB_CRYPT_IV_LEN;
134
135 /* plaintext length as little-endian uint64 */
136 uint64_t ptlen_le = (uint64_t)sqlite_len;
137 unsigned char ptlen_buf[8];
138 for (int i = 0; i < 8; i++)
139 ptlen_buf[i] = (unsigned char)(ptlen_le >> (8 * i));
140 memcpy(blob + off, ptlen_buf, 8); off += 8;
141
142 memcpy(blob + off, ct, (size_t)ct_len);
143
144 free(ct);
145
146 *out_len = total;
147 return blob;
148}

150unsigned char *cwist_db_crypt_open(const cwist_db_crypt_ctx_t *ctx,
151 const unsigned char *blob,
152 size_t blob_len,
153 size_t *out_len) {
154 if (!ctx || !blob || !out_len) return NULL;
155
156 /* Minimum header: 4+1+16+64+16+8 = 109 bytes. */
157 if (blob_len < CWIST_DB_CRYPT_HDR_LEN) return NULL;
158
159 size_t off = 0;
160
161 /* 1. Verify magic and version. */
162 if (memcmp(blob + off, MAGIC, 4) != 0) return NULL;
163 off += 4;
164 if (blob[off] != VERSION) return NULL;
165 off += 1;
166
167 /* 2. Read KEK IV. */
168 const unsigned char *kek_iv = blob + off;
169 off += CWIST_DB_CRYPT_IV_LEN;
170
171 /* 3. Decrypt DEK blob (48 bytes padded to 64 during seal).
172 * The encrypted DEK blob occupies:
173 * enc_dek_len = blob_len - fixed_parts_after_enc_dek
174 * fixed_parts_after: dek_iv(16) + ptlen(8) + at_least_one_block(16)
175 * We know enc_dek_len = (blob_len - off - 16 - 8 - ct_len)
176 * but we stored enc_dek_len as 48+16=64 bytes always.
177 * Recalculate: plain was 48 bytes → padded = 64. */
178 const size_t enc_dek_len = 64;
179 if (blob_len < off + enc_dek_len + CWIST_DB_CRYPT_IV_LEN + 8) return NULL;
180
181 unsigned char dek_plain[CWIST_DB_CRYPT_IV_LEN + CWIST_DB_CRYPT_KEY_LEN + 16];
182 int dek_plain_len = aes256_decrypt(ctx->kek, kek_iv,
183 blob + off, enc_dek_len,
184 dek_plain);
185 if (dek_plain_len < (int)(CWIST_DB_CRYPT_IV_LEN + CWIST_DB_CRYPT_KEY_LEN))
186 return NULL;
187 off += enc_dek_len;
188
189 const unsigned char *dek_iv = dek_plain;
190 const unsigned char *dek = dek_plain + CWIST_DB_CRYPT_IV_LEN;
191
192 /* 4. Read stored DEK IV (should match what was unwrapped). */
193 off += CWIST_DB_CRYPT_IV_LEN; /* skip dek_iv in header (redundant; we use the unwrapped one) */
194
195 /* 5. Read plaintext length (little-endian uint64). */
196 uint64_t ptlen_le = 0;
197 for (int i = 0; i < 8; i++)
198 ptlen_le |= ((uint64_t)blob[off + i]) << (8 * i);
199 off += 8;
200 size_t expected_pt_len = (size_t)ptlen_le;
201
202 /* 6. Decrypt SQLite bytes. */
203 size_t ct_len = blob_len - off;
204 if (ct_len == 0) return NULL;
205
206 unsigned char *pt = (unsigned char *)malloc(ct_len + 1); /* extra for safety */
207 if (!pt) return NULL;
208
209 int pt_len = aes256_decrypt(dek, dek_iv, blob + off, ct_len, pt);
210 if (pt_len < 0 || (size_t)pt_len < expected_pt_len) {
211 free(pt);
212 return NULL;
213 }
214
215 pt[expected_pt_len] = '\0'; /* null-terminate for safety (sqlite doesn't need it) */
216 *out_len = expected_pt_len;
217 return pt;
218}
