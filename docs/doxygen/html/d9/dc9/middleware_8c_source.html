27static char *generate_request_id() {
28 static const char charset[] = "abcdefghijklmnopqrstuvwxyz0123456789";
29 char *id = cwist_alloc(17);
30
31 pthread_mutex_lock(&rid_mutex);
32 if (rid_seed == 0) rid_seed = (unsigned int)time(NULL) ^ (unsigned int)pthread_self();
33 unsigned int seed = rid_seed++;
34 pthread_mutex_unlock(&rid_mutex);
35
36 for (int i = 0; i < 16; i++) {
37 id[i] = charset[rand_r(&seed) % (sizeof(charset) - 1)];
38 }
39 id[16] = '\0';
40 return id;
41}

129#define MAX_IP_TRACK 1024

131static int ip_cache_count = 0;

140void cwist_mw_rate_limit_ip_handler(cwist_http_request *req, cwist_http_response *res, cwist_handler_func next) {
141
142 // Get client ip from fd
143 cwist_sstring *ip = cwist_get_client_ip_from_fd(req->client_fd);
144
145 time_t now = time(NULL);
146 ip_limit_t *found = NULL;
147
148 pthread_mutex_lock(&rate_mutex);
149
150 // if ip is found on ip_cache, set found = &ip_cache[i];
151 for (int i = 0; i < ip_cache_count; i++) {
152 if (!cwist_sstring_compare(ip, ip_cache[i].ip)) {
153 found = &ip_cache[i];
154 break;
155 }
156 }
157
158 // if not found, add current ip to ip_cache
159 if (!found && ip_cache_count < MAX_IP_TRACK) {
160 found = &ip_cache[ip_cache_count++];
161 strncpy(found->ip, ip->data, sizeof(found->ip) - 1);
162 found->last_reset = now;
163 found->count = 0;
164 }
165
166 // if found, refresh reset time
167 if (found) {
168 if (now - found->last_reset >= 60) {
169 found->last_reset = now;
170 found->count = 0;
171 }
172
173 // if cound count is more than 60, block request
174 if (found->count >= 60) {
175 pthread_mutex_unlock(&rate_mutex);
176 res->status_code = 429;
177 cwist_sstring_assign(res->body, "Too Many Requests");
178 cwist_http_header_add(&res->headers, "Retry-After", "60");
179 return;
180 }
181 found->count++;
182 }
183 cwist_sstring_destroy(ip);
184 pthread_mutex_unlock(&rate_mutex);
185
186 next(req, res);
187}

235/*
236 * We use req->private_data to carry the decoded claims across the middleware
237 * boundary into the downstream handler. The previous value of private_data
238 * is saved/restored so other middleware can also use that field without
239 * conflict.
240 *
241 * The context is stack-allocated inside each wrapper function. It is only
242 * valid during the synchronous execution of the middleware chain; no pointer
243 * to the context escapes to asynchronous code.
244 */
245
246#define CWIST_JWT_CTX_MAGIC 0x4A574354UL /* "JWCT" */
247
248typedef struct {
249 unsigned long magic;
250 const char *secret;
251 cwist_jwt_claims *claims;
252 void *prev_private_data;
253} cwist_jwt_ctx_t;

261void cwist_mw_jwt_auth_handler(cwist_http_request *req, cwist_http_response *res, cwist_handler_func next) {
262 /* Retrieve the secret stored in the context tag */
263 cwist_jwt_ctx_t *ctx = (cwist_jwt_ctx_t *)req->private_data;
264 if (!ctx) {
265 /* Should not happen if the middleware was set up correctly */
266 res->status_code = CWIST_HTTP_INTERNAL_ERROR;
267 cwist_sstring_assign(res->body, "{\"error\":\"JWT middleware misconfigured\"}");
268 cwist_http_header_add(&res->headers, "Content-Type", "application/json");
269 return;
270 }
271
272 const char *secret = ctx->secret;
273
274 /* Extract Bearer token from Authorization header */
275 char *auth_header = cwist_http_header_get(req->headers, "Authorization");
276 if (!auth_header) {
277 res->status_code = CWIST_HTTP_UNAUTHORIZED;
278 cwist_sstring_assign(res->body, "{\"error\":\"Missing Authorization header\"}");
279 cwist_http_header_add(&res->headers, "Content-Type", "application/json");
280 return;
281 }
282
283 /* Expect "Bearer <token>" */
284 if (strncmp(auth_header, "Bearer ", 7) != 0) {
285 res->status_code = CWIST_HTTP_UNAUTHORIZED;
286 cwist_sstring_assign(res->body, "{\"error\":\"Invalid Authorization scheme\"}");
287 cwist_http_header_add(&res->headers, "Content-Type", "application/json");
288 return;
289 }
290
291 const char *token = auth_header + 7;
292
293 cwist_jwt_claims *claims = cwist_jwt_verify(token, secret);
294 if (!claims) {
295 res->status_code = CWIST_HTTP_UNAUTHORIZED;
296 cwist_sstring_assign(res->body, "{\"error\":\"Invalid or expired token\"}");
297 cwist_http_header_add(&res->headers, "Content-Type", "application/json");
298 return;
299 }
300
301 /* Stash claims so downstream handlers can retrieve them */
302 ctx->claims = claims;
303
304 next(req, res);
305
306 /* Clean up after the chain returns */
307 cwist_jwt_claims_destroy(claims);
308 ctx->claims = NULL;
309}

311/*
312 * Factory — we use a small heap-allocated context to bind the secret to the
313 * handler. Because cwist_middleware_func is a plain function pointer we cannot
314 * capture the secret in a closure, so we embed it in the req->private_data
315 * field before dispatching.
316 *
317 * The returned function pointer is cwist_mw_jwt_auth_handler. The secret is
318 * stored in a per-request cwist_jwt_ctx_t that is pushed/popped around the
319 * call so it does not trample any existing private_data.
320 */
321
322typedef struct {
323 const char *secret;
324} cwist_jwt_mw_cfg_t;
325
326/* We keep a small static table of registered secrets. A server rarely needs
327 * more than a handful of distinct signing keys. 8 slots cover the common case
328 * (e.g. one per audience / service) without dynamic allocation. If more are
329 * needed, increase CWIST_JWT_MAX_SECRETS and add the corresponding wrapper. */
330#define CWIST_JWT_MAX_SECRETS 8
331
332static cwist_jwt_mw_cfg_t s_jwt_cfgs[CWIST_JWT_MAX_SECRETS];
333static int s_jwt_cfg_count = 0;
334static pthread_mutex_t s_jwt_cfg_mutex = PTHREAD_MUTEX_INITIALIZER;
335
336/* One wrapper function per registered secret slot.
337 * The ctx is stack-allocated; its lifetime is confined to this function frame. */
338#define CWIST_JWT_DEFINE_WRAPPER(N) \
339static void cwist_mw_jwt_wrap_##N(cwist_http_request *req, cwist_http_response *res, cwist_handler_func next) { \
340 cwist_jwt_ctx_t ctx = { \
341 .magic = CWIST_JWT_CTX_MAGIC, \
342 .secret = s_jwt_cfgs[N].secret, \
343 .claims = NULL, \
344 .prev_private_data = req->private_data \
345 }; \
346 req->private_data = &ctx; \
347 cwist_mw_jwt_auth_handler(req, res, next); \
348 req->private_data = ctx.prev_private_data; \
349}

370cwist_middleware_func cwist_mw_jwt_auth(const char *secret) {
371 if (!secret) return NULL;
372
373 pthread_mutex_lock(&s_jwt_cfg_mutex);
374
375 /* Reuse existing slot for the same secret pointer */
376 for (int i = 0; i < s_jwt_cfg_count; i++) {
377 if (s_jwt_cfgs[i].secret == secret) {
378 pthread_mutex_unlock(&s_jwt_cfg_mutex);
379 return s_jwt_wrappers[i];
380 }
381 }
382
383 if (s_jwt_cfg_count >= CWIST_JWT_MAX_SECRETS) {
384 pthread_mutex_unlock(&s_jwt_cfg_mutex);
385 fprintf(stderr, "[CWIST] cwist_mw_jwt_auth: maximum secret slots exceeded\n");
386 return NULL;
387 }
388
389 int slot = s_jwt_cfg_count++;
390 s_jwt_cfgs[slot].secret = secret;
391
392 pthread_mutex_unlock(&s_jwt_cfg_mutex);
393 return s_jwt_wrappers[slot];
394}
