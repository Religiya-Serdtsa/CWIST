69static void cwist_owner_lock_init(void) {
70 if (atomic_load_explicit(&g_owner_lock_ready, memory_order_acquire)) {
71 return;
72 }
73 static atomic_flag init_flag = ATOMIC_FLAG_INIT;
74 while (atomic_flag_test_and_set_explicit(&init_flag, memory_order_acquire)) {
75 // Spin until initialization completes; extremely rare path.
76 }
77 if (!atomic_load_explicit(&g_owner_lock_ready, memory_order_relaxed)) {
78 ttak_mutex_init(&g_owner_lock);
79 atomic_store_explicit(&g_owner_lock_ready, true, memory_order_release);
80 }
81 atomic_flag_clear_explicit(&init_flag, memory_order_release);
82}

141ttak_owner_t *cwist_create_owner(void) {
142 cwist_owner_lock_init();
143 ttak_mutex_lock(&g_owner_lock);
144 if (!g_owner) {
145 ttak_owner_t *owner = ttak_owner_create(TTAK_OWNER_SAFE_DEFAULT | TTAK_OWNER_DENY_DANGEROUS_MEM);
146 if (owner) {
147 bool ok = true;
148 ok &= ttak_owner_register_resource(owner, CWIST_OWNER_RESOURCE, &g_owner_policy);
149 ok &= ttak_owner_register_func(owner, CWIST_OWNER_ALLOC_FUNC, cwist_owner_alloc);
150 ok &= ttak_owner_register_func(owner, CWIST_OWNER_FREE_FUNC, cwist_owner_free);
151 ok &= ttak_owner_register_func(owner, CWIST_OWNER_REALLOC_FUNC, cwist_owner_realloc);
152 if (!ok) {
153 ttak_owner_destroy(owner);
154 owner = NULL;
155 fprintf(stderr, "[CWIST] Failed to register owner guards\n");
156 }
157 } else {
158 fprintf(stderr, "[CWIST] Failed to create owner context\n");
159 }
160 g_owner = owner;
161 }
162 ttak_owner_t *owner = g_owner;
163 ttak_mutex_unlock(&g_owner_lock);
164 return owner;
165}

173static bool cwist_owner_call(const char *func_name, void *args) {
174 ttak_owner_t *owner = cwist_create_owner();
175 if (!owner) {
