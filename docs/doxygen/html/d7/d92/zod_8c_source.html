65cwist_zod_result_t cwist_zod_validate(const cJSON *json, const cwist_schema_t *schema) {
66 cwist_zod_result_t r;
67 memset(&r, 0, sizeof(r));
68 r.valid = true;
69
70 if (!json || !schema) {
71 add_error(&r, "(root)", "null json or schema pointer");
72 return r;
73 }
74
75 if (!cJSON_IsObject(json)) {
76 add_error(&r, "(root)", "expected a JSON object at root");
77 return r;
78 }
79
80 for (int fi = 0; fi < schema->field_count; fi++) {
81 const cwist_schema_field_t *fd = &schema->fields[fi];
82 const cJSON *item = cJSON_GetObjectItemCaseSensitive(json, fd->name);
83
84 /* Missing field */
85 if (!item) {
86 if (fd->required) {
87 char msg[CWIST_ZOD_MSG_MAX];
88 snprintf(msg, sizeof(msg),
89 "required field '%s' is missing", fd->name);
90 add_error(&r, fd->name, msg);
91 }
92 continue;
93 }
94
95 /* Wrong type */
96 if (!type_matches(item, fd->type)) {
97 char msg[CWIST_ZOD_MSG_MAX];
98 snprintf(msg, sizeof(msg),
99 "expected type '%s'", field_type_name(fd->type));
100 add_error(&r, fd->name, msg);
101 }
102 }
103
104 return r;
105}
