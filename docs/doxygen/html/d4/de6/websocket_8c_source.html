32cwist_websocket *cwist_websocket_upgrade(cwist_http_request *req, int client_fd) {
33 if (!req || client_fd < 0) return NULL;
34
35 // Validate Headers
36 char *connection = cwist_http_header_get(req->headers, "Connection");
37 char *upgrade = cwist_http_header_get(req->headers, "Upgrade");
38 char *key = cwist_http_header_get(req->headers, "Sec-WebSocket-Key");
39
40 if (!connection || !upgrade || !key) return NULL;
41 if (strcasestr(connection, "Upgrade") == NULL) return NULL;
42 if (strcasecmp(upgrade, "websocket") != 0) return NULL;
43
44 // Handshake Key Generation
45 char combined_key[512];
46 snprintf(combined_key, sizeof(combined_key), "%s%s", key, WS_GUID);
47
48 uint8_t hash[20];
49 sha1((uint8_t *)combined_key, strlen(combined_key), hash);
50
51 size_t accept_len;
52 char *accept_key = base64_encode(hash, 20, &accept_len);
53 if (!accept_key) return NULL;
54
55 // Send Response
56 cwist_http_response *res = cwist_http_response_create();
57 res->status_code = 101;
58 cwist_sstring_assign(res->status_text, "Switching Protocols");
59 cwist_http_header_add(&res->headers, "Upgrade", "websocket");
60 cwist_http_header_add(&res->headers, "Connection", "Upgrade");
61 cwist_http_header_add(&res->headers, "Sec-WebSocket-Accept", accept_key);
62
63 cwist_error_t err = cwist_http_send_response(client_fd, res);
64
65 cwist_http_response_destroy(res);
66 cwist_free(accept_key);
67
68 if (err.error.err_i16 != 0) return NULL;
69
70 req->upgraded = true;
71
72 cwist_websocket *ws = (cwist_websocket *)cwist_alloc(sizeof(cwist_websocket));
73 ws->fd = client_fd;
74 ws->is_closed = false;
75 return ws;
76}

105cwist_ws_frame *cwist_websocket_receive(cwist_websocket *ws) {
106 if (!ws || ws->is_closed) return NULL;
107
108 uint8_t head[2];
109 if (read_exact(ws->fd, head, 2) < 0) return NULL;
110
111 bool fin = (head[0] & 0x80) != 0;
112 cwist_ws_opcode_t opcode = head[0] & 0x0F;
113 bool masked = (head[1] & 0x80) != 0;
114 uint64_t payload_len = head[1] & 0x7F;
115
116 if (!masked) {
117 // Client-to-server frames must be masked per spec
118 // We can choose to strict close or allow. Strict is better.
119 // For now, let's just return error.
120 return NULL;
121 }
122
123 if (payload_len == 126) {
124 uint16_t len16;
125 if (read_exact(ws->fd, &len16, 2) < 0) return NULL;
126 payload_len = ntohs(len16);
127 } else if (payload_len == 127) {
128 uint64_t len64;
129 if (read_exact(ws->fd, &len64, 8) < 0) return NULL;
130 // manually swap if no be64toh
131 // assuming be64toh or similar exists, or manual
132 // Linux usually has be64toh in <endian.h>
133 // Let's implement manual swap to be portable
134 uint8_t *p = (uint8_t *)&len64;
135 payload_len = ((uint64_t)p[0] << 56) | ((uint64_t)p[1] << 48) |
136 ((uint64_t)p[2] << 40) | ((uint64_t)p[3] << 32) |
137 ((uint64_t)p[4] << 24) | ((uint64_t)p[5] << 16) |
138 ((uint64_t)p[6] << 8) | ((uint64_t)p[7]);
139 }
140
141 uint8_t masking_key[4];
142 if (read_exact(ws->fd, masking_key, 4) < 0) return NULL;
143
144 uint8_t *payload = NULL;
145 if (payload_len > 0) {
146 payload = (uint8_t *)cwist_alloc(payload_len + 1); // +1 for safety null term if text
147 if (!payload) return NULL;
148 if (read_exact(ws->fd, payload, payload_len) < 0) {
149 cwist_free(payload);
150 return NULL;
151 }
152
153 // Unmask
154 for (uint64_t i = 0; i < payload_len; i++) {
155 payload[i] ^= masking_key[i % 4];
156 }
157 payload[payload_len] = '\0'; // Null terminate for convenience if text
158 }
159
160 if (opcode == CWIST_WS_FRAME_CLOSE) {
161 ws->is_closed = true;
162 }
163
164 cwist_ws_frame *frame = (cwist_ws_frame *)cwist_alloc(sizeof(cwist_ws_frame));
165 frame->fin = fin;
166 frame->opcode = opcode;
167 frame->payload = payload;
168 frame->payload_len = payload_len;
169
170 return frame;
171}

181int cwist_websocket_send(cwist_websocket *ws, cwist_ws_opcode_t opcode, const uint8_t *data, size_t len) {
182 if (!ws || ws->is_closed) return -1;
183
184 uint8_t head[10]; // Max header size (2 + 8)
185 size_t head_len = 2;
186
187 head[0] = 0x80 | (opcode & 0x0F); // FIN=1
188
189 if (len < 126) {
190 head[1] = len;
191 } else if (len < 65536) {
192 head[1] = 126;
193 head[2] = (len >> 8) & 0xFF;
194 head[3] = len & 0xFF;
195 head_len += 2;
196 } else {
197 head[1] = 127;
198 for (int i = 0; i < 8; i++) {
199 head[2 + i] = (len >> ((7 - i) * 8)) & 0xFF;
200 }
201 head_len += 8;
202 }
203
204 // Server does not mask frames
205
206 if (send(ws->fd, head, head_len, 0) < 0) return -1;
207 if (len > 0) {
208 if (send(ws->fd, data, len, 0) < 0) return -1;
209 }
210
211 return 0;
212}
