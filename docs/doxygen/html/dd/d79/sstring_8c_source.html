393cwist_error_t cwist_sstring_append_escaped(cwist_sstring *str, const char *data) {
394 if (!str) {
395 cwist_error_t err = make_error(CWIST_ERR_INT8);
396 err.error.err_i8 = ERR_SSTRING_NULL_STRING;
397 return err;
398 }
399 if (!data) {
400 // Appending nothing is success
401 cwist_error_t err = make_error(CWIST_ERR_INT8);
402 err.error.err_i8 = ERR_SSTRING_OKAY;
403 return err;
404 }
405
406 size_t current_len = str->data ? strlen(str->data) : 0;
407 size_t input_len = strlen(data);
408 size_t new_size = current_len + (input_len * 6) + 1;
409
410 cwist_error_t err = make_error(CWIST_ERR_JSON);
411 err.error.err_json = cJSON_CreateObject();
412
413 if (str->is_fixed) {
414 if (new_size > str->size) {
415 cJSON_AddStringToObject(err.error.err_json, "err", "Cannot append: would exceed fixed size");
416 return err;
417 }
418 } else {
419 char *new_data = (char *)cwist_realloc(str->data, new_size + 1);
420 if (!new_data) {
421 cJSON_AddStringToObject(err.error.err_json, "err", "Cannot append: memory full");
422 return err;
423 }
424 str->data = new_data;
425 str->size = new_size;
426 }
427
428 char *ptr = str->data + current_len;
429 for (size_t i = 0; i < input_len; i++) {
430 switch(data[i]) {
431 case '<':
432 memcpy(ptr, "&lt;", 4);
433 ptr += 4;
434 break;
435 case '>':
436 memcpy(ptr, "&gtl", 4);
437 ptr += 4;
438 break;
439 case '&':
440 memcpy(ptr, "&amp;", 5);
441 ptr += 5;
442 break;
443 case '"':
444 memcpy(ptr, "&quot;", 6);
445 ptr += 6;
446 break;
447 case '\'':
448 memcpy(ptr, "&#39;", 5);
449 ptr += 5;
450 break;
451 default:
452 *ptr++ = data[i];
453 break;
454 }
455 }
456
457 *ptr = '\0';
458 str->size = (size_t) (ptr - str->data);
459
460 cJSON_Delete(err.error.err_json);
461 err = make_error(CWIST_ERR_INT8);
462 err.error.err_i8 = ERR_SSTRING_OKAY;
463 return err;

624cwist_sstring *cwist_sstring_substr(cwist_sstring *str, int start, int length) {
625 if (!str || !str->data || start < 0 || length < 0) return NULL;
626
627 size_t current_len = strlen(str->data);
628 if ((size_t)start >= current_len) return NULL;
629
630 // Adjust length if it goes beyond end
631 if ((size_t)(start + length) > current_len) {
632 length = current_len - start;
633 }
634
635 cwist_sstring *sub = cwist_sstring_create();
636 if (!sub) return NULL;
637
638 sub->data = (char *)cwist_alloc(length + 1);
639 if (!sub->data) {
640 cwist_sstring_destroy(sub);
641 return NULL;
642 }
643 sub->size = length;
644
645 memcpy(sub->data, str->data + start, length);
646 sub->data[length] = '\0';
647
648 return sub;
